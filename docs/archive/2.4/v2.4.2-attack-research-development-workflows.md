# v2.4.2 Enhancement: ATT&CK Research & Development Workflows

**Version:** 2.4.2  
**Status:** Planning  
**Priority:** High  
**Dependencies:** v2.3.6.1 (Vulnerability Investigation), Tool Registry, OffSec Team R&D

## Executive Summary

This enhancement establishes a **complete, automated end-to-end workflow** from vulnerability discovery through research, POC development, tool production, testing, and deployment into ATT&CK-based operational workflows. It bridges 6 critical gaps identified in the current system, transforming the R&D Lab from a tracking system into an active development pipeline.

### Current State Problems

1. **R&D Experiments are created but never executed** - No orchestration layer
2. **POC artifacts have no storage mechanism** - Generated code disappears
3. **Tool registration is manual and disconnected** - No R&D → Tool Registry automation
4. **Tool testing is isolated** - No feedback loop for refinement
5. **ATT&CK workflows lack tool integration** - Can't deploy validated tools operationally
6. **Nuclei template creation is incomplete** - Investigation loop is broken

### Proposed Solution

A **7-phase automated pipeline** that takes a vulnerability from discovery to operational deployment:

```
Discovery → Investigation → R&D Research → POC Development → Tool Production 
   → Testing & Validation → ATT&CK Workflow Deployment → Feedback Loop
```

## Critical Gap Solutions

### Gap 1: R&D Experiment Execution

**Problem:** Experiments created as "planned" but never executed.

**Solution:** R&D Experiment Orchestrator service

```typescript
/**
 * server/services/rd-experiment-orchestrator.ts
 * 
 * Orchestrates the execution of R&D experiments across three phases:
 * 1. Research Phase - Deep vulnerability analysis
 * 2. POC Development Phase - Exploit/tool code generation
 * 3. Nuclei Template Phase - Detection template creation
 */

import { EventEmitter } from 'events';
import { db } from '../db';
import { rdExperiments, researchProjects, rdArtifacts, vulnerabilities } from '@shared/schema';
import { eq, and } from 'drizzle-orm';

// ============================================================================
// Types & Interfaces
// ============================================================================

export interface ExperimentExecutionContext {
  experimentId: string;
  projectId: string;
  vulnerabilityId: string;
  operationId: string;
  targetInfo?: {
    id: string;
    name: string;
    value: string;
    type: string;
  };
}

export interface ResearchArtifact {
  type: 'research_document';
  title: string;
  content: string;
  findings: {
    exploitationVectors: string[];
    prerequisites: string[];
    attackComplexity: 'low' | 'medium' | 'high';
    detectionDifficulty: 'low' | 'medium' | 'high';
    references: string[];
  };
  metadata: {
    cveAnalysis?: string;
    cweMapping?: string;
    attackSurface?: string[];
  };
}

export interface POCArtifact {
  type: 'poc_code';
  language: 'python' | 'ruby' | 'bash' | 'powershell' | 'javascript';
  sourceCode: string;
  filename: string;
  dependencies: string[];
  usage: string;
  reliability: 'high' | 'medium' | 'low';
  evasionTechniques?: string[];
  metadata: {
    targetPlatform: string[];
    payloadType?: string;
    deliveryMethod?: string;
  };
}

export interface NucleiTemplateArtifact {
  type: 'nuclei_template';
  templateId: string;
  yamlContent: string;
  severity: 'critical' | 'high' | 'medium' | 'low' | 'informational';
  matchers: Array<{
    type: string;
    condition?: string;
    part?: string;
    words?: string[];
    regex?: string[];
  }>;
  extractors?: Array<{
    type: string;
    part?: string;
    regex?: string[];
    group?: number;
  }>;
  metadata: {
    verified: boolean;
    falsePositiveRate?: number;
  };
}

export type Artifact = ResearchArtifact | POCArtifact | NucleiTemplateArtifact;

export interface ExperimentResult {
  experimentId: string;
  success: boolean;
  artifacts: Artifact[];
  executionLog: string[];
  errors?: string[];
  startedAt: Date;
  completedAt: Date;
  durationMs: number;
}

// ============================================================================
// R&D Experiment Orchestrator
// ============================================================================

class RDExperimentOrchestrator extends EventEmitter {
  private activeExecutions: Map<string, AbortController> = new Map();

  /**
   * Execute a complete R&D experiment (all phases)
   */
  async executeExperiment(
    experimentId: string,
    context: ExperimentExecutionContext
  ): Promise<ExperimentResult> {
    const startTime = Date.now();
    const executionLog: string[] = [];
    const artifacts: Artifact[] = [];
    const errors: string[] = [];

    // Check if already executing
    if (this.activeExecutions.has(experimentId)) {
      throw new Error(`Experiment ${experimentId} is already executing`);
    }

    const abortController = new AbortController();
    this.activeExecutions.set(experimentId, abortController);

    try {
      executionLog.push(`[${new Date().toISOString()}] Starting experiment execution`);

      // Update experiment status to 'running'
      await db
        .update(rdExperiments)
        .set({
          status: 'running',
          startedAt: new Date(),
        })
        .where(eq(rdExperiments.id, experimentId));

      this.emit('experiment_started', { experimentId, context });

      // Fetch experiment details
      const [experiment] = await db
        .select()
        .from(rdExperiments)
        .where(eq(rdExperiments.id, experimentId));

      if (!experiment) {
        throw new Error(`Experiment ${experimentId} not found`);
      }

      // Determine experiment type based on name pattern
      const experimentName = experiment.name.toLowerCase();
      let artifact: Artifact | null = null;

      if (experimentName.includes('research')) {
        executionLog.push('Executing Research Phase...');
        artifact = await this.executeResearchPhase(experiment, context, executionLog);
      } else if (experimentName.includes('poc')) {
        executionLog.push('Executing POC Development Phase...');
        artifact = await this.executePOCPhase(experiment, context, executionLog);
      } else if (experimentName.includes('nuclei')) {
        executionLog.push('Executing Nuclei Template Phase...');
        artifact = await this.executeNucleiPhase(experiment, context, executionLog);
      } else {
        throw new Error(`Unknown experiment type: ${experiment.name}`);
      }

      if (artifact) {
        artifacts.push(artifact);
        executionLog.push(`Generated artifact: ${artifact.type}`);

        // Store artifact in database
        await this.storeArtifact(experimentId, artifact, executionLog);
      }

      // Update experiment with results
      await db
        .update(rdExperiments)
        .set({
          status: 'completed',
          completedAt: new Date(),
          results: {
            artifacts: artifacts.map(a => ({ type: a.type })),
            executionLog,
          },
        })
        .where(eq(rdExperiments.id, experimentId));

      executionLog.push(`[${new Date().toISOString()}] Experiment completed successfully`);

      const result: ExperimentResult = {
        experimentId,
        success: true,
        artifacts,
        executionLog,
        startedAt: new Date(startTime),
        completedAt: new Date(),
        durationMs: Date.now() - startTime,
      };

      this.emit('experiment_completed', result);

      return result;
    } catch (error) {
      const errorMsg = error instanceof Error ? error.message : 'Unknown error';
      errors.push(errorMsg);
      executionLog.push(`[ERROR] ${errorMsg}`);

      // Update experiment status to failed
      await db
        .update(rdExperiments)
        .set({
          status: 'failed',
          completedAt: new Date(),
          errorMessage: errorMsg,
          results: { executionLog, errors },
        })
        .where(eq(rdExperiments.id, experimentId));

      this.emit('experiment_failed', { experimentId, error: errorMsg });

      return {
        experimentId,
        success: false,
        artifacts,
        executionLog,
        errors,
        startedAt: new Date(startTime),
        completedAt: new Date(),
        durationMs: Date.now() - startTime,
      };
    } finally {
      this.activeExecutions.delete(experimentId);
    }
  }

  /**
   * Execute Research Phase - Deep vulnerability analysis
   */
  private async executeResearchPhase(
    experiment: any,
    context: ExperimentExecutionContext,
    log: string[]
  ): Promise<ResearchArtifact> {
    log.push('Delegating to R&D Team Agent for research...');

    // Import the R&D Team Agent
    const { rdTeamAgent } = await import('./agents/rd-team-agent');

    // Ensure agent is initialized
    if (!rdTeamAgent.isInitialized) {
      await rdTeamAgent.initialize();
    }

    // Fetch vulnerability details
    const [vuln] = await db
      .select()
      .from(vulnerabilities)
      .where(eq(vulnerabilities.id, context.vulnerabilityId));

    if (!vuln) {
      throw new Error(`Vulnerability ${context.vulnerabilityId} not found`);
    }

    // Execute research task
    const result = await rdTeamAgent.executeTask({
      taskType: 'vulnerability_research',
      taskName: `Deep Research: ${vuln.title}`,
      description: experiment.description,
      operationId: context.operationId,
      parameters: {
        vulnerabilityId: context.vulnerabilityId,
        cveId: vuln.cveId,
        cweId: vuln.cweId,
        title: vuln.title,
        description: vuln.description,
        severity: vuln.severity,
        targetInfo: context.targetInfo,
        researchObjectives: experiment.objectives || experiment.hypothesis,
      },
    });

    if (!result.success) {
      throw new Error(`Research phase failed: ${result.error}`);
    }

    log.push('Research phase completed successfully');

    // Transform agent result into ResearchArtifact
    const artifact: ResearchArtifact = {
      type: 'research_document',
      title: `Vulnerability Research: ${vuln.title}`,
      content: result.data?.researchDocument || result.output || '',
      findings: {
        exploitationVectors: result.data?.exploitationVectors || [],
        prerequisites: result.data?.prerequisites || [],
        attackComplexity: result.data?.attackComplexity || 'medium',
        detectionDifficulty: result.data?.detectionDifficulty || 'medium',
        references: result.data?.references || [],
      },
      metadata: {
        cveAnalysis: result.data?.cveAnalysis,
        cweMapping: result.data?.cweMapping,
        attackSurface: result.data?.attackSurface || [],
      },
    };

    return artifact;
  }

  /**
   * Execute POC Development Phase - Generate exploit code
   */
  private async executePOCPhase(
    experiment: any,
    context: ExperimentExecutionContext,
    log: string[]
  ): Promise<POCArtifact> {
    log.push('Delegating to POC Development Agent...');

    // Import POC Development Agent (to be created)
    const { pocDevelopmentAgent } = await import('./agents/poc-development-agent');

    if (!pocDevelopmentAgent.isInitialized) {
      await pocDevelopmentAgent.initialize();
    }

    // Fetch research artifacts from previous phase
    const researchArtifacts = await db
      .select()
      .from(rdArtifacts)
      .where(
        and(
          eq(rdArtifacts.projectId, context.projectId),
          eq(rdArtifacts.artifactType, 'research_document')
        )
      );

    const researchContext = researchArtifacts.length > 0
      ? researchArtifacts[0].content
      : experiment.description;

    // Execute POC development task
    const result = await pocDevelopmentAgent.executeTask({
      taskType: 'poc_development',
      taskName: experiment.name,
      description: experiment.description,
      operationId: context.operationId,
      parameters: {
        vulnerabilityId: context.vulnerabilityId,
        researchContext,
        targetInfo: context.targetInfo,
        methodology: experiment.methodology,
      },
    });

    if (!result.success) {
      throw new Error(`POC development failed: ${result.error}`);
    }

    log.push('POC development completed successfully');

    const artifact: POCArtifact = {
      type: 'poc_code',
      language: result.data?.language || 'python',
      sourceCode: result.data?.sourceCode || result.output || '',
      filename: result.data?.filename || 'exploit.py',
      dependencies: result.data?.dependencies || [],
      usage: result.data?.usage || '# No usage instructions provided',
      reliability: result.data?.reliability || 'medium',
      evasionTechniques: result.data?.evasionTechniques || [],
      metadata: {
        targetPlatform: result.data?.targetPlatform || ['linux'],
        payloadType: result.data?.payloadType,
        deliveryMethod: result.data?.deliveryMethod,
      },
    };

    return artifact;
  }

  /**
   * Execute Nuclei Template Phase - Generate detection template
   */
  private async executeNucleiPhase(
    experiment: any,
    context: ExperimentExecutionContext,
    log: string[]
  ): Promise<NucleiTemplateArtifact> {
    log.push('Delegating to Nuclei Template Agent...');

    // Import Nuclei Template Agent (to be created)
    const { nucleiTemplateAgent } = await import('./agents/nuclei-template-agent');

    if (!nucleiTemplateAgent.isInitialized) {
      await nucleiTemplateAgent.initialize();
    }

    // Fetch POC artifacts from previous phase
    const pocArtifacts = await db
      .select()
      .from(rdArtifacts)
      .where(
        and(
          eq(rdArtifacts.projectId, context.projectId),
          eq(rdArtifacts.artifactType, 'poc_code')
        )
      );

    const pocContext = pocArtifacts.length > 0
      ? pocArtifacts[0].content
      : experiment.description;

    // Execute template generation task
    const result = await nucleiTemplateAgent.executeTask({
      taskType: 'nuclei_template_generation',
      taskName: experiment.name,
      description: experiment.description,
      operationId: context.operationId,
      parameters: {
        vulnerabilityId: context.vulnerabilityId,
        pocContext,
        targetInfo: context.targetInfo,
        methodology: experiment.methodology,
      },
    });

    if (!result.success) {
      throw new Error(`Nuclei template generation failed: ${result.error}`);
    }

    log.push('Nuclei template generated successfully');

    const artifact: NucleiTemplateArtifact = {
      type: 'nuclei_template',
      templateId: result.data?.templateId || `custom-${context.vulnerabilityId.substring(0, 8)}`,
      yamlContent: result.data?.yamlContent || result.output || '',
      severity: result.data?.severity || 'medium',
      matchers: result.data?.matchers || [],
      extractors: result.data?.extractors,
      metadata: {
        verified: false,
        falsePositiveRate: result.data?.falsePositiveRate,
      },
    };

    return artifact;
  }

  /**
   * Store artifact in rd_artifacts table
   */
  private async storeArtifact(
    experimentId: string,
    artifact: Artifact,
    log: string[]
  ): Promise<void> {
    log.push(`Storing artifact: ${artifact.type}`);

    // Get experiment and project details
    const [experiment] = await db
      .select()
      .from(rdExperiments)
      .where(eq(rdExperiments.id, experimentId));

    if (!experiment) {
      throw new Error(`Experiment ${experimentId} not found`);
    }

    let content: string;
    let language: string | undefined;
    let filename: string | undefined;

    switch (artifact.type) {
      case 'research_document':
        content = artifact.content;
        language = 'markdown';
        filename = 'research.md';
        break;
      case 'poc_code':
        content = artifact.sourceCode;
        language = artifact.language;
        filename = artifact.filename;
        break;
      case 'nuclei_template':
        content = artifact.yamlContent;
        language = 'yaml';
        filename = `${artifact.templateId}.yaml`;
        break;
    }

    await db.insert(rdArtifacts).values({
      experimentId,
      projectId: experiment.projectId,
      artifactType: artifact.type,
      content,
      language,
      filename,
      metadata: artifact,
      createdAt: new Date(),
    });

    log.push(`Artifact stored successfully: ${filename}`);
  }

  /**
   * Execute all experiments for a research project sequentially
   */
  async executeProject(projectId: string): Promise<{
    projectId: string;
    totalExperiments: number;
    completed: number;
    failed: number;
    results: ExperimentResult[];
  }> {
    // Fetch all experiments for this project
    const experiments = await db
      .select()
      .from(rdExperiments)
      .where(eq(rdExperiments.projectId, projectId))
      .orderBy(rdExperiments.createdAt);

    const results: ExperimentResult[] = [];
    let completed = 0;
    let failed = 0;

    // Fetch project details for context
    const [project] = await db
      .select()
      .from(researchProjects)
      .where(eq(researchProjects.id, projectId));

    if (!project) {
      throw new Error(`Research project ${projectId} not found`);
    }

    const context: ExperimentExecutionContext = {
      experimentId: '', // Set per experiment
      projectId,
      vulnerabilityId: project.sourceVulnerabilityId || '',
      operationId: '', // From vulnerability
    };

    // Execute experiments sequentially
    for (const experiment of experiments) {
      if (experiment.status !== 'planned') {
        continue; // Skip already executed experiments
      }

      context.experimentId = experiment.id;

      try {
        const result = await this.executeExperiment(experiment.id, context);
        results.push(result);

        if (result.success) {
          completed++;
        } else {
          failed++;
        }
      } catch (error) {
        failed++;
        console.error(`Failed to execute experiment ${experiment.id}:`, error);
      }
    }

    // Update project status
    const allComplete = completed + failed === experiments.length;
    if (allComplete) {
      await db
        .update(researchProjects)
        .set({
          status: failed === 0 ? 'completed' : 'active',
          completedAt: failed === 0 ? new Date() : undefined,
        })
        .where(eq(researchProjects.id, projectId));
    }

    return {
      projectId,
      totalExperiments: experiments.length,
      completed,
      failed,
      results,
    };
  }

  /**
   * Cancel an actively running experiment
   */
  async cancelExperiment(experimentId: string): Promise<boolean> {
    const controller = this.activeExecutions.get(experimentId);
    if (!controller) {
      return false;
    }

    controller.abort();
    this.activeExecutions.delete(experimentId);

    await db
      .update(rdExperiments)
      .set({
        status: 'cancelled',
        completedAt: new Date(),
      })
      .where(eq(rdExperiments.id, experimentId));

    return true;
  }

  /**
   * Get execution status
   */
  isExecuting(experimentId: string): boolean {
    return this.activeExecutions.has(experimentId);
  }

  get activeExecutionCount(): number {
    return this.activeExecutions.size;
  }
}

// Singleton instance
export const rdExperimentOrchestrator = new RDExperimentOrchestrator();
```

### Gap 2: POC/Tool Artifact Storage

**Problem:** No file storage for generated POC code and artifacts.

**Solution:** rd_artifacts table + file system storage

#### Database Schema Addition

```sql
-- Add to db/migrations/0025_rd_workflow_automation.sql

-- Artifact storage for R&D experiment outputs
CREATE TABLE IF NOT EXISTS rd_artifacts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  experiment_id UUID NOT NULL REFERENCES rd_experiments(id) ON DELETE CASCADE,
  project_id UUID NOT NULL REFERENCES research_projects(id) ON DELETE CASCADE,
  
  -- Artifact metadata
  artifact_type TEXT NOT NULL CHECK (artifact_type IN ('research_document', 'poc_code', 'nuclei_template', 'tool_binary')),
  filename TEXT,
  language TEXT, -- python, ruby, yaml, markdown, bash, powershell
  
  -- Content storage
  content TEXT, -- Source code or document text
  file_path TEXT, -- Path if stored on filesystem
  file_size INTEGER, -- Size in bytes
  file_hash TEXT, -- SHA256 hash for integrity
  
  -- Metadata
  metadata JSONB DEFAULT '{}'::jsonb,
  tags TEXT[] DEFAULT '{}',
  
  -- Audit
  created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_rd_artifacts_experiment ON rd_artifacts(experiment_id);
CREATE INDEX idx_rd_artifacts_project ON rd_artifacts(project_id);
CREATE INDEX idx_rd_artifacts_type ON rd_artifacts(artifact_type);
```

### Gap 3: Tool Registration from R&D Output

**Problem:** Manual, disconnected tool registration process.

**Solution:** Automated tool promotion workflow

```typescript
/**
 * server/services/rd-tool-promotion.ts
 * 
 * Automatically promotes R&D POC artifacts to the Tool Registry
 * with full metadata extraction and ATT&CK mapping.
 */

import { db } from '../db';
import { rdArtifacts, toolRegistry, toolRegistryTactics, toolRegistryTechniques, vulnerabilities, attackTactics, attackTechniques } from '@shared/schema';
import { eq, and } from 'drizzle-orm';
import type { ToolConfiguration } from '@shared/types/tool-config';
import type { POCArtifact } from './rd-experiment-orchestrator';

// ============================================================================
// Types
// ============================================================================

export interface ToolPromotionRequest {
  artifactId: string;
  toolName?: string;
  category?: string;
  overrideMetadata?: Partial<ToolConfiguration>;
}

export interface ToolPromotionResult {
  success: boolean;
  toolId?: string;
  toolRegistryId?: string;
  error?: string;
  metadata: {
    artifactId: string;
    extractedMetadata: Partial<ToolConfiguration>;
    attackMappings: {
      tactics: string[];
      techniques: string[];
    };
  };
}

// ============================================================================
// R&D Tool Promotion Service
// ============================================================================

class RDToolPromotion {
  /**
   * Promote a POC artifact to the Tool Registry
   */
  async promoteToToolRegistry(
    request: ToolPromotionRequest
  ): Promise<ToolPromotionResult> {
    try {
      // Fetch the artifact
      const [artifact] = await db
        .select()
        .from(rdArtifacts)
        .where(eq(rdArtifacts.id, request.artifactId));

      if (!artifact) {
        return {
          success: false,
          error: `Artifact ${request.artifactId} not found`,
          metadata: {
            artifactId: request.artifactId,
            extractedMetadata: {},
            attackMappings: { tactics: [], techniques: [] },
          },
        };
      }

      if (artifact.artifactType !== 'poc_code') {
        return {
          success: false,
          error: `Can only promote poc_code artifacts. Got: ${artifact.artifactType}`,
          metadata: {
            artifactId: request.artifactId,
            extractedMetadata: {},
            attackMappings: { tactics: [], techniques: [] },
          },
        };
      }

      // Extract metadata from POC artifact
      const pocMetadata = artifact.metadata as POCArtifact;
      const extractedMetadata = await this.extractToolMetadata(artifact, pocMetadata);

      // Merge with user overrides
      const finalMetadata: ToolConfiguration = {
        ...extractedMetadata,
        ...request.overrideMetadata,
      };

      // Determine tool category
      const category = request.category || this.inferCategory(pocMetadata, artifact.filename || '');

      // Generate unique tool ID
      const toolId = request.toolName?.toLowerCase().replace(/[^a-z0-9-]/g, '-') ||
        this.generateToolId(artifact.filename || 'custom-exploit');

      // Create Tool Registry entry
      const [tool] = await db.insert(toolRegistry).values({
        toolId,
        name: request.toolName || artifact.filename || 'Custom R&D Tool',
        version: '1.0.0-rd',
        category,
        description: `R&D-generated tool from experiment ${artifact.experimentId}`,
        
        installMethod: 'manual',
        binaryPath: `/opt/rd-tools/${artifact.filename}`,
        
        config: finalMetadata,
        
        containerName: 'rtpi-tools',
        containerUser: 'rtpi-tools',
        
        installStatus: 'pending',
        validationStatus: 'untested',
        
        tags: ['r&d-generated', ...(pocMetadata.evasionTechniques || [])],
        metadata: {
          rdArtifactId: artifact.id,
          rdExperimentId: artifact.experimentId,
          rdProjectId: artifact.projectId,
          language: artifact.language,
          reliability: pocMetadata.reliability,
        },
        
        createdAt: new Date(),
        updatedAt: new Date(),
      }).returning();

      // Auto-assign ATT&CK mappings
      const attackMappings = await this.assignATTACKMapping(tool.id, artifact);

      return {
        success: true,
        toolId: tool.toolId,
        toolRegistryId: tool.id,
        metadata: {
          artifactId: request.artifactId,
          extractedMetadata: finalMetadata,
          attackMappings,
        },
      };
    } catch (error) {
      const errorMsg = error instanceof Error ? error.message : 'Unknown error';
      return {
        success: false,
        error: errorMsg,
        metadata: {
          artifactId: request.artifactId,
          extractedMetadata: {},
          attackMappings: { tactics: [], techniques: [] },
        },
      };
    }
  }

  /**
   * Extract tool metadata from POC artifact
   */
  private async extractToolMetadata(
    artifact: any,
    pocMetadata: POCArtifact
  ): Promise<Partial<ToolConfiguration>> {
    const sourceCode = artifact.content as string;

    // Parse command-line arguments from source code
    const parameters = this.parseParameters(sourceCode, pocMetadata.language);

    // Build tool configuration
    const config: Partial<ToolConfiguration> = {
      toolId: '', // Set by caller
      name: '', // Set by caller
      description: `Exploit tool for vulnerability detection and exploitation`,
      category: 'exploitation',
      binaryPath: `/opt/rd-tools/${artifact.filename}`,
      
      parameters,
      
      outputParser: {
        parserType: 'regex',
        regexPatterns: this.generateOutputPatterns(pocMetadata),
      },
      
      tags: ['r&d', 'custom', ...(pocMetadata.evasionTechniques || [])],
      
      metadata: {
        language: pocMetadata.language,
        targetPlatform: pocMetadata.metadata.targetPlatform,
        reliability: pocMetadata.reliability,
        dependencies: pocMetadata.dependencies,
      },
    };

    return config;
  }

  /**
   * Parse parameters from source code
   */
  private parseParameters(sourceCode: string, language: string): any[] {
    // Simple parameter extraction from common patterns
    const parameters: any[] = [];

    if (language === 'python') {
      // Look for argparse patterns
      const argParseRegex = /add_argument\(['"](--?\w+)['"],.*?help=['"]([^'"]+)['"]/g;
      let match;
      while ((match = argParseRegex.exec(sourceCode)) !== null) {
        parameters.push({
          name: match[1].replace(/^--?/, ''),
          type: 'string',
          description: match[2],
          required: !match[1].startsWith('--'),
        });
      }
    } else if (language === 'bash') {
      // Look for getopts patterns or usage functions
      const usageRegex = /\s+(-\w)\s+<(\w+)>\s+-\s+(.+)/g;
      let match;
      while ((match = usageRegex.exec(sourceCode)) !== null) {
        parameters.push({
          name: match[2],
          type: 'string',
          description: match[3],
          required: true,
        });
      }
    }

    // Fallback: basic target parameter
    if (parameters.length === 0) {
      parameters.push({
        name: 'target',
        type: 'string',
        description: 'Target URL or IP address',
        required: true,
      });
    }

    return parameters;
  }

  /**
   * Generate output parsing patterns
   */
  private generateOutputPatterns(pocMetadata: POCArtifact): Record<string, string[]> {
    return {
      success: ['vulnerable', 'exploit successful', 'pwned', 'shell opened'],
      failure: ['not vulnerable', 'failed', 'error', 'connection refused'],
      data: ['(?<data>[A-Za-z0-9+/=]{20,})'], // Base64-like strings
    };
  }

  /**
   * Infer tool category from metadata
   */
  private inferCategory(pocMetadata: POCArtifact, filename: string): string {
    const combinedText = `${filename} ${JSON.stringify(pocMetadata.metadata)}`.toLowerCase();

    if (/sql|sqli/i.test(combinedText)) return 'exploitation';
    if (/xss|cross-site/i.test(combinedText)) return 'web-application';
    if (/rce|remote.*code/i.test(combinedText)) return 'exploitation';
    if (/scanner|detect/i.test(combinedText)) return 'scanning';
    if (/password|crack/i.test(combinedText)) return 'password-cracking';

    return 'exploitation'; // Default
  }

  /**
   * Generate unique tool ID
   */
  private generateToolId(filename: string): string {
    const base = filename.replace(/\.[^.]+$/, '').toLowerCase();
    const timestamp = Date.now().toString(36);
    return `rd-${base}-${timestamp}`;
  }

  /**
   * Auto-assign ATT&CK tactic and technique mappings
   */
  private async assignATTACKMapping(
    toolRegistryId: string,
    artifact: any
  ): Promise<{ tactics: string[]; techniques: string[] }> {
    // Get vulnerability associated with this artifact's project
    const [experiment] = await db
      .select()
      .from(rdExperiments)
      .where(eq(rdExperiments.id, artifact.experimentId));

    if (!experiment) {
      return { tactics: [], techniques: [] };
    }

    const [project] = await db
      .select()
      .from(researchProjects)
      .where(eq(researchProjects.id, experiment.projectId));

    if (!project || !project.sourceVulnerabilityId) {
      return { tactics: [], techniques: [] };
    }

    const [vuln] = await db
      .select()
      .from(vulnerabilities)
      .where(eq(vulnerabilities.id, project.sourceVulnerabilityId));

    if (!vuln) {
      return { tactics: [], techniques: [] };
    }

    // Infer ATT&CK mappings from vulnerability type
    const mappings = this.inferATTACKMappings(vuln);

    // Assign tactics
    const tactics = await this.assignTactics(toolRegistryId, mappings.tacticNames);

    // Assign techniques
    const techniques = await this.assignTechniques(toolRegistryId, mappings.techniqueIds);

    return {
      tactics: mappings.tacticNames,
      techniques: mappings.techniqueIds,
    };
  }

  /**
   * Infer ATT&CK tactics and techniques from vulnerability
   */
  private inferATTACKMappings(vuln: any): {
    tacticNames: string[];
    techniqueIds: string[];
  } {
    const title = vuln.title.toLowerCase();
    const description = (vuln.description || '').toLowerCase();
    const combined = `${title} ${description}`;

    const tacticNames: string[] = [];
    const techniqueIds: string[] = [];

    // Reconnaissance
    if (/recon|discovery|enumeration/i.test(combined)) {
      tacticNames.push('Reconnaissance');
      techniqueIds.push('T1595'); // Active Scanning
    }

    // Initial Access
    if (/exploit|vulnerability|rce|injection/i.test(combined)) {
      tacticNames.push('Initial Access');
      techniqueIds.push('T1190'); // Exploit Public-Facing Application
    }

    // Execution
    if (/rce|command.*inject|code.*exec/i.test(combined)) {
      tacticNames.push('Execution');
      techniqueIds.push('T1059'); // Command and Scripting Interpreter
    }

    // Credential Access
    if (/password|credential|auth.*bypass/i.test(combined)) {
      tacticNames.push('Credential Access');
      techniqueIds.push('T1110'); // Brute Force
    }

    // Discovery
    if (/discover|enumerate|scan/i.test(combined)) {
      tacticNames.push('Discovery');
      techniqueIds.push('T1046'); // Network Service Scanning
    }

    // Default: Reconnaissance if nothing else matches
    if (tacticNames.length === 0) {
      tacticNames.push('Reconnaissance');
      techniqueIds.push('T1595');
    }

    return { tacticNames, techniqueIds };
  }

  /**
   * Assign tactics to tool
   */
  private async assignTactics(
    toolRegistryId: string,
    tacticNames: string[]
  ): Promise<void> {
    for (const tacticName of tacticNames) {
      const [tactic] = await db
        .select()
        .from(attackTactics)
        .where(eq(attackTactics.name, tacticName))
        .limit(1);

      if (tactic) {
        // Check if mapping already exists
        const existing = await db
          .select()
          .from(toolRegistryTactics)
          .where(
            and(
              eq(toolRegistryTactics.toolId, toolRegistryId),
              eq(toolRegistryTactics.tacticId, tactic.id)
            )
          );

        if (existing.length === 0) {
          await db.insert(toolRegistryTactics).values({
            toolId: toolRegistryId,
            tacticId: tactic.id,
            createdAt: new Date(),
          });
        }
      }
    }
  }

  /**
   * Assign techniques to tool
   */
  private async assignTechniques(
    toolRegistryId: string,
    techniqueIds: string[]
  ): Promise<void> {
    for (const attackId of techniqueIds) {
      const [technique] = await db
        .select()
        .from(attackTechniques)
        .where(eq(attackTechniques.attackId, attackId))
        .limit(1);

      if (technique) {
        // Check if mapping already exists
        const existing = await db
          .select()
          .from(toolRegistryTechniques)
          .where(
            and(
              eq(toolRegistryTechniques.toolId, toolRegistryId),
              eq(toolRegistryTechniques.techniqueId, technique.id)
            )
          );

        if (existing.length === 0) {
          await db.insert(toolRegistryTechniques).values({
            toolId: toolRegistryId,
            techniqueId: technique.id,
            createdAt: new Date(),
          });
        }
      }
    }
  }
}
```

```typescript
import { Router } from "express";

    res.status(500).json({ error: error.message });

  const { id } = req.params;
```

```typescript
import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Card } from '@/components/ui/card';
import { Play, StopCircle, CheckCircle, XCircle, Loader2 } from 'lucide-react';
import { api } from '@/lib/api';
import { toast } from 'sonner';

interface ExperimentExecutionPanelProps {
  experimentId: string;
  projectId: string;
  vulnerabilityId: string;
  operationId: string;
  onComplete?: () => void;
}

export function ExperimentExecutionPanel({
  experimentId,
  projectId,
  vulnerabilityId,
  operationId,
  onComplete,
}: ExperimentExecutionPanelProps) {
  const [executing, setExecuting] = useState(false);
  const [result, setResult] = useState<any>(null);
  const [log, setLog] = useState<string[]>([]);

  const handleExecute = async () => {
    setExecuting(true);
    setLog([]);
    
    try {
      const response = await api.post(`/rd-workflow-automation/experiments/${experimentId}/execute`, {
        projectId,
        vulnerabilityId,
        operationId,
      });
      
      setResult(response);
      setLog(response.executionLog || []);
      
      if (response.success) {
        toast.success('Experiment completed successfully');
        onComplete?.();
      } else {
        toast.error('Experiment failed');
      }
    } catch (error) {
      toast.error('Failed to execute experiment');
    } finally {
      setExecuting(false);
    }
  };

  return (
    <Card className="p-6">
      <div className="flex items-center justify-between mb-4">
        <h3 className="text-lg font-semibold">Experiment Execution</h3>
        
        <Button
          onClick={handleExecute}
          disabled={executing}
          variant={executing ? "secondary" : "default"}
        >
          {executing ? (
            <>
              <Loader2 className="h-4 w-4 mr-2 animate-spin" />
              Executing...
            </>
          ) : (
            <>
              <Play className="h-4 w-4 mr-2" />
              Execute
            </>
          )}
        </Button>
      </div>

      {/* Execution Log */}
      {log.length > 0 && (
        <div className="bg-slate-900 text-green-400 p-4 rounded-lg font-mono text-sm max-h-96 overflow-y-auto">
          {log.map((line, i) => (
            <div key={i}>{line}</div>
          ))}
        </div>
      )}

      {/* Result Summary */}
      {result && (
        <div className="mt-4 p-4 border rounded-lg">
          <div className="flex items-center gap-2 mb-2">
            {result.success ? (
              <CheckCircle className="h-5 w-5 text-green-600" />
            ) : (
              <XCircle className="h-5 w-5 text-red-600" />
            )}
            <span className="font-semibold">
              {result.success ? 'Success' : 'Failed'}
            </span>
          </div>
          
          <div className="text-sm text-muted-foreground">
            <div>Artifacts Generated: {result.artifacts?.length || 0}</div>
            <div>Duration: {(result.durationMs / 1000).toFixed(2)}s</div>
          </div>
        </div>
      )}
    </Card>
  );
```
