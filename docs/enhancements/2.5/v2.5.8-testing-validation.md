### 8.2 Integration Test Scenarios

**Test 1: BOF Execution with Staged I/O**
```rust
#[tokio::test]
async fn test_bof_execution_with_output() {
    let (tx, mut rx) = mpsc::channel(32);
    let runtime = SandboxRuntime::new(tx).unwrap();

    let bof_bytes = include_bytes!("fixtures/test_bof.o");
    let args = BofArgument::new().add_str("test_arg").pack();

    let result = runtime.execute(
        ExecutableFormat::Bof,
        bof_bytes,
        Capabilities::bof_defaults(),
        [("args".into(), args)].into(),
        vec![],
    ).await.unwrap();

    assert!(result.completed);
    assert_eq!(result.exit_code, 0);
    assert!(!result.stdout.is_empty());

    // Verify staged data was sent through broker
    let staged = rx.try_recv().unwrap();
    assert_eq!(staged.channel, "beacon_output");
}
```

### 8.3 Security Test Cases

```rust
#[tokio::test]
async fn test_capability_denial_filesystem() {
    // BOF with no fs_read_paths tries to read /etc/passwd
    let result = runtime.execute(
        ExecutableFormat::Bof, &read_file_bof,
        Capabilities::bof_defaults(), // no fs access
        HashMap::new(), vec![],
    ).await.unwrap();
    // Should fail with CapabilityDenied, not return file contents
    assert!(result.error.as_ref().unwrap().contains("denied"));
}

#[tokio::test]
async fn test_memory_limit_enforcement() {
    let caps = Capabilities { max_memory_bytes: 1024, ..Default::default() };
    // Payload that tries to allocate 1MB
    let result = runtime.execute(
        ExecutableFormat::Shellcode, &alloc_shellcode,
        caps, HashMap::new(), vec![],
    ).await.unwrap();
    assert!(!result.completed);
    assert!(result.error.as_ref().unwrap().contains("exhausted"));
}

#[tokio::test]
async fn test_cpu_time_enforcement() {
    let caps = Capabilities { max_cpu_time_ms: 100, ..Default::default() };
    // Infinite loop shellcode
    let result = runtime.execute(
        ExecutableFormat::Shellcode, &infinite_loop,
        caps, HashMap::new(), vec![],
    ).await.unwrap();
    assert!(!result.completed);
}

#[tokio::test]
async fn test_path_traversal_blocked() {
    let caps = Capabilities {
        fs_read_paths: vec!["/tmp".into()],
        ..Default::default()
    };
    // Try to read /etc/passwd via ../../etc/passwd
    assert!(!caps.can_read_path("/tmp/../../etc/passwd"));
}

#[tokio::test]
async fn test_fork_denied_in_elf() {
    // ELF that calls fork() syscall
    let result = runtime.execute(
        ExecutableFormat::Elf, &fork_elf,
        Capabilities::recon_defaults(), HashMap::new(), vec![],
    ).await.unwrap();
    // fork should return -1 (EPERM)
}
```

### 8.4 Test Fixtures

Create minimal test binaries:

| Fixture | Format | Purpose |
|---------|--------|---------|
| `test_bof.o` | COFF | Simple BOF that calls BeaconPrintf("Hello from BOF") |
| `test_pe.exe` | PE x64 | Minimal PE that writes to stdout and exits |
| `test_elf` | ELF x64 | Static hello-world binary |
| `test.wasm` | WASM | Rust hello-world compiled to wasm32-wasi |
| `infinite_loop.bin` | Shellcode | `jmp $` for timeout testing |
| `alloc_bomb.bin` | Shellcode | Attempts large mmap for memory limit testing |

Build scripts for fixtures:
```bash
# BOF: compile with mingw
x86_64-w64-mingw32-gcc -c test_bof.c -o test_bof.o

# PE: compile with mingw
x86_64-w64-mingw32-gcc -o test_pe.exe test_pe.c -nostdlib

# ELF: compile static
gcc -static -o test_elf test_elf.c

# WASM: compile with Rust
cargo build --target wasm32-wasi --release -p test-wasm-fixture
```
