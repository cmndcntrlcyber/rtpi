# v2.6.4 — Native Binary Loaders (PE/ELF)

## Overview

Implement loaders for native executables: Windows PE (x64) and Linux ELF (x64). Both use Unicorn CPU emulation for maximum isolation. This phase also implements the OS-specific syscall/API emulation layers that allow native binaries to interact with the host through the broker.

use goblin::pe::PE;

pub fn load_pe(&self, data: &[u8], emu: &mut Unicorn<EmulatorContext>) -> SandboxResult<u64> {
    let pe = PE::parse(data).map_err(|e| SandboxError::InvalidFormat(e.to_string()))?;

    // 1. Map sections with correct permissions
    for section in &pe.sections {

**TEB/PEB Setup (Windows emulation context):**
```rust
fn setup_teb_peb(emu: &mut Unicorn<EmulatorContext>) -> SandboxResult<()> {
    // Map TEB at 0x7FF00000 (64KB)
    emu.mem_map(TEB_BASE, TEB_SIZE, Permission::READ | Permission::WRITE)?;

    // TEB structure (key fields):
    // +0x000 SEH chain (pointer to self for simple programs)
    // +0x008 Stack base
    // +0x010 Stack limit
    // +0x030 PEB pointer
    // +0x048 TLS array

    let peb_addr = TEB_BASE + 0x1000;
    // PEB fields: ImageBaseAddress, Ldr, ProcessParameters

    // Set GS segment base to TEB (x64 Windows uses GS for TEB)
    emu.reg_write(RegisterX86::GS_BASE, TEB_BASE)?;

    Ok(())
}
```

### 4.2 ELF Loader (`elf.rs`)

**Parse with goblin:**
```rust
use goblin::elf::Elf;

pub fn load_elf(&self, data: &[u8], emu: &mut Unicorn<EmulatorContext>) -> SandboxResult<u64> {
    let elf = Elf::parse(data).map_err(|e| SandboxError::InvalidFormat(e.to_string()))?;

    // 1. Load PT_LOAD segments
    for phdr in elf.program_headers.iter().filter(|p| p.p_type == PT_LOAD) {
        let perms = elf_permissions(phdr.p_flags);
        emu.mem_map(phdr.p_vaddr, align_up(phdr.p_memsz as usize), perms)?;
        emu.mem_write(phdr.p_vaddr, &data[phdr.p_offset as usize..(phdr.p_offset + phdr.p_filesz) as usize])?;
    }

    // 2. Setup initial stack: [argc, argv[0..n], NULL, envp[0..m], NULL, auxv...]
    setup_elf_stack(emu, &args, &envp)?;

    // 3. Build auxiliary vector (auxv)
    push_auxv(emu, AT_ENTRY, elf.entry)?;
    push_auxv(emu, AT_PHDR, phdr_addr)?;
    push_auxv(emu, AT_PHNUM, elf.program_headers.len())?;
    push_auxv(emu, AT_PAGESZ, 4096)?;
    push_auxv(emu, AT_NULL, 0)?;

    // 4. Return entry point
    Ok(elf.entry)
}
```

### 4.4 Linux Syscall Emulation (`emulation/linux.rs`)

Hook `syscall` instruction and dispatch:

```rust
fn handle_syscall(emu: &mut Unicorn<EmulatorContext>) {
    let syscall_num = emu.reg_read(RegisterX86::RAX).unwrap();
    match syscall_num {
        // Standard Linux syscalls (emulated)
        0 => handle_read(emu),      // read(fd, buf, count)
        1 => handle_write(emu),     // write(fd, buf, count) → broker
        3 => handle_close(emu),     // close(fd)
        9 => handle_mmap(emu),      // mmap → broker AllocateMemory
        12 => handle_brk(emu),      // brk → heap extension
        60 => handle_exit(emu),     // exit → stop emulation
        231 => handle_exit(emu),    // exit_group

        // Custom sandbox syscalls (0x2000+)
        0x2000 => handle_stage_output(emu),   // StageOutput → broker
        0x2001 => handle_get_input(emu),      // GetInput → broker
        0x2002 => handle_host_request(emu),   // HostRequest → broker
        0x2003 => handle_alloc_heap(emu),     // AllocHeap → broker
        0x2004 => handle_get_context(emu),    // GetContext → execution metadata

        // Denied syscalls
        57 | 58 | 59 => {  // fork, vfork, execve
            emu.reg_write(RegisterX86::RAX, (-1i64) as u64).unwrap();
        }

        _ => {
            log::warn!("Unhandled syscall: {}", syscall_num);
            emu.reg_write(RegisterX86::RAX, (-38i64) as u64).unwrap(); // -ENOSYS
        }
    }
}
```
