### 6.2 Server-Side Service Implementation

```rust
// nexus-infra/src/grpc_server.rs

pub struct SandboxService {
    sandbox_runtime: Arc<SandboxRuntime>,
    staged_data_rx: Arc<Mutex<mpsc::Receiver<StagedData>>>,
}

#[tonic::async_trait]
impl NexusSandbox for SandboxService {
    type ExecuteInSandboxStream = ReceiverStream<Result<SandboxExecuteResult, Status>>;

    async fn execute_in_sandbox(
        &self,
        request: Request<SandboxExecuteTask>,
    ) -> Result<Response<Self::ExecuteInSandboxStream>, Status> {
        let task = request.into_inner();
        let (tx, rx) = mpsc::channel(32);

        let runtime = self.sandbox_runtime.clone();
        tokio::spawn(async move {
            let format = proto_to_format(task.format());
            let capabilities = proto_to_capabilities(&task.capabilities);
            let input_data = task.staged_input;

            let result = runtime.execute(
                format, &task.payload, capabilities, input_data, task.args
            ).await;

            let proto_result = match result {
                Ok(r) => execution_result_to_proto(&task.task_id, &r),
                Err(e) => error_to_proto_result(&task.task_id, &e),
            };

            let _ = tx.send(Ok(proto_result)).await;
        });

        Ok(Response::new(ReceiverStream::new(rx)))
    }

    // StreamStagedData — bidirectional streaming for real-time data exfiltration
    // PushStagedInput — push input data before/during execution
    // GetSandboxStatus — query active execution contexts
}
```

### 6.3 Client-Side Methods

```rust
// nexus-infra/src/grpc_client.rs

impl GrpcClient {
    pub async fn execute_in_sandbox(
        &mut self,
        task: SandboxExecuteTask,
    ) -> Result<Vec<SandboxExecuteResult>, InfraError> {
        let mut stream = self.sandbox_client
            .execute_in_sandbox(Request::new(task))
            .await?
            .into_inner();

        let mut results = Vec::new();
        while let Some(result) = stream.message().await? {
            results.push(result);
        }
        Ok(results)
    }
}
```

### 6.4 Type Conversions

```rust
fn proto_to_format(format: i32) -> ExecutableFormat {
    match format {
        1 => ExecutableFormat::Bof,
        2 => ExecutableFormat::Pe,
        3 => ExecutableFormat::Elf,
        4 => ExecutableFormat::Wasm,
        5 => ExecutableFormat::Shellcode,
        _ => ExecutableFormat::Bof, // default
    }
}

fn proto_to_capabilities(proto: &Option<proto::Capabilities>) -> Capabilities {
    match proto {
        Some(c) => Capabilities {
            allowed_channels: c.allowed_channels.iter().cloned().collect(),
            max_memory_bytes: c.max_memory_bytes as usize,
            max_cpu_time_ms: c.max_cpu_time_ms,
            // ...
        },
        None => Capabilities::bof_defaults(),
    }
}
```
