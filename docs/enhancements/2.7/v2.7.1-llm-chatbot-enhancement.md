**Created:** 2026-03-11

## Overview

Enhance the existing Operations Manager floating chat to include full LLM chatbot capabilities, allowing users to:
1. Ask questions about operations directly to the Operations Manager AI
2. Provide instructions and receive intelligent responses
3. Access individual agent chat windows for specialized interactions
4. Maintain protected, context-aware conversations with each agent

```typescript
async function ensureAgentChatAccess(
  req: Request,
  res: Response,
  next: NextFunction
) {
  const { agentId } = req.params;
  const user = req.user as any;
  
  // Check permissions
  const hasAccess = await canUserAccessAgentChat(
    user.id,
    agentId,
    req.query.conversationId
  );
  
  if (!hasAccess) {
    return res.status(403).json({ error: "Access denied to agent chat" });
  }
  
  next();
}
```

## Database Schema

```sql
-- Migration: 0026_add_agent_conversations.sql

CREATE TYPE message_role AS ENUM ('user', 'agent', 'system');
CREATE TYPE message_status AS ENUM ('sending', 'delivered', 'error');

CREATE TABLE agent_conversations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  agent_id UUID NOT NULL REFERENCES agents(id) ON DELETE CASCADE,
  operation_id UUID REFERENCES operations(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  
  conversation_name TEXT DEFAULT 'Untitled Conversation',
  is_active BOOLEAN DEFAULT true,
  is_protected BOOLEAN DEFAULT true,
  allowed_user_ids UUID[] DEFAULT ARRAY[]::UUID[],
  
  -- Context snapshot
  context_data JSON DEFAULT '{}',
  
  -- Encryption
  is_encrypted BOOLEAN DEFAULT false,
  encryption_key_id TEXT,
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  last_message_at TIMESTAMP
);

CREATE TABLE agent_conversation_messages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  conversation_id UUID NOT NULL REFERENCES agent_conversations(id) ON DELETE CASCADE,
  
  role message_role NOT NULL,
  content TEXT NOT NULL,
  
  -- Attribution
  user_id UUID REFERENCES users(id) ON DELETE SET NULL,
  agent_id UUID REFERENCES agents(id) ON DELETE SET NULL,
  
  -- AI metadata
  ai_provider TEXT,
  ai_model TEXT,
  tokens_used INTEGER,
  temperature REAL,
  
  -- Memory integration
  stored_as_memory_id UUID REFERENCES memory_entries(id) ON DELETE SET NULL,
  relevant_memory_ids UUID[],
  
  -- Attachments
  attachments JSON DEFAULT '[]',
  
  -- Status
  status message_status DEFAULT 'delivered',
  error_message TEXT,
  
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_conv_agent_op ON agent_conversations(agent_id, operation_id);
CREATE INDEX idx_conv_user ON agent_conversations(user_id);
CREATE INDEX idx_conv_messages_conv ON agent_conversation_messages(conversation_id);
CREATE INDEX idx_conv_messages_created ON agent_conversation_messages(created_at DESC);

-- View for easy querying
CREATE VIEW agent_conversation_summary AS
SELECT 
  c.id,
  c.conversation_name,
  a.name as agent_name,
  a.role as agent_role,
  u.username,
  o.name as operation_name,
  COUNT(m.id) as message_count,
  MAX(m.created_at) as last_message_at,
  c.is_active,
  c.is_protected
FROM agent_conversations c
LEFT JOIN agents a ON c.agent_id = a.id
LEFT JOIN users u ON c.user_id = u.id
LEFT JOIN operations o ON c.operation_id = o.id
LEFT JOIN agent_conversation_messages m ON m.conversation_id = c.id
GROUP BY c.id, a.name, a.role, u.username, o.name;
```
